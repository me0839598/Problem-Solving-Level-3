#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define dd double
int main(){
     // MAS Programming
    int n;
    cin>>n;
    string s;
    cin>>s;
    map<char,int> mp;
    for(int i=0;i<n;i++)
    {
        s[i]=tolower(s[i]);
        mp[s[i]]++;
    }
    if(mp.size()==26)
        cout<<"YES";
    else cout<<"NO";
    return 0;
}
